Q-5: Consider the iris dataset and plot a pie chart describing Iris Species Distribution.

Q-9: Take the iris dataset as input and plot each columnwithHistogram.

In[]:
import pandas as pd
import matplotlib.pyplot as plt

file_path='DSD Datasets/Iris.csv'
df = pd.read_csv(file_path)
# Selecting only the numerical columns (excluding 'Id' and 'Species')
numeric_columns = df.drop(columns=['Id', 'Species'])
# Plotting histograms for each numerical column
for i, column in enumerate(numeric_columns.columns, 1):
    plt.subplot(2, 2, i)  
    plt.hist(numeric_columns[column], bins=20, edgecolor='black', alpha=0.7)
    plt.title(f'Histogram of {column}')
    plt.xlabel(column)
    plt.ylabel('Frequency')

plt.tight_layout()
plt.show()
